'use strict';

const fs = require('fs');

// COPY THIS REGEX VERBATIM — do not modify
const TAG_LINE_RE = /^[ \t]*(?:\/\/+|\/\*+|\*+|#+|--+|"{3}|'{3})[ \t]*@gsd-(\w+)(?:\(([^)]*)\))?[ \t]*(.*?)[ \t]*$/gm;

**`parseMetadata(raw)` helper:** Takes the captured group 2 string (e.g., `"phase:2, priority:high"`). Returns `{}` if falsy. Otherwise split on `/,\s*/`, then each part on `:` (max 2 parts). Returns `{ key: value, ... }`.

**`scanFile(filePath)` function:**
- `const content = fs.readFileSync(filePath, 'utf-8')`
- Reset TAG_LINE_RE.lastIndex = 0 before each use (important — gm flag is stateful)
- Create a new regex each call using `new RegExp(TAG_LINE_RE.source, 'gm')` to avoid lastIndex state issues
- Use the regex with `[...content.matchAll(re)]`
- For each match: compute line number by counting `\n` chars in `content.slice(0, match.index)` plus 1
- Build tag object: `{ type: match[1], file: filePath, line, metadata: parseMetadata(match[2]), description: (match[3] || '').trim(), raw: match[0].trim() }`
- Only include tags where `VALID_TAG_TYPES.has(type)` (filter out unknown types)
- Return array of tag objects

**`scanDirectory(dirPath, options)` function:**
- `options = { phaseFilter, typeFilter, excludes }` — all optional
- Walk with `fs.readdirSync(dir, { withFileTypes: true })`
- Skip entries where `DEFAULT_EXCLUDES.includes(entry.name)` (directories) or `options.excludes` contains the name
- Also check for `.gsdignore` at `path.join(dirPath, '.gsdignore')` if it exists — read its lines, skip dirs/files matching any line
- Recurse into subdirectories, call scanFile on files
- After collecting all tags: if `phaseFilter` provided, filter to `tag.metadata.phase === String(phaseFilter)`
- If `typeFilter` provided, filter to `tag.type === typeFilter`
- Return array of all matching tags

**`formatAsJson(tags)` function:** `return JSON.stringify(tags, null, 2);`

**`formatAsMarkdown(tags, projectName)` function:**
Generate CODE-INVENTORY.md string. Structure:
- Header with generated timestamp, project name, schema version 1.0, total tag count and file count
- `## Summary Statistics` — markdown table: Tag Type | Count (one row per type that appears in tags)
- `## Tags by Type` — for each type that has tags: H3 `### @gsd-<type>`, then for each file with that type: H4 `#### <relative-file-path>`, then markdown table: Line | Metadata | Description
- `## Phase Reference Index` — table: Phase | Tag Count | Files (grouped by metadata.phase value; use "(untagged)" for tags with no phase key)

**`cmdExtractTags(cwd, targetPath, options)` function:**
- `options = { phaseFilter, typeFilter, format, outputFile }`
- `format` defaults to `'json'`
- Call `scanDirectory(path.resolve(cwd, targetPath || cwd), { phaseFilter: options.phaseFilter, typeFilter: options.typeFilter })`
- If `format === 'md'` and `options.outputFile`: `fs.mkdirSync(path.dirname(options.outputFile), { recursive: true })` then write markdown
- Otherwise write to process.stdout

**At bottom:** `module.exports = { scanFile, scanDirectory, formatAsJson, formatAsMarkdown, cmdExtractTags };`
