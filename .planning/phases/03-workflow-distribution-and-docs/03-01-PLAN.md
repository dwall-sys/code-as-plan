**gsd-tools.cjs change:** Add a new `case 'set-mode'` branch in the main switch statement. Use this exact implementation:

    case 'set-mode': {
      const allArgs = args.slice(1);
      const named = parseNamedArgs(allArgs, ['phase']);
      const modeValue = allArgs.find(a => !a.startsWith('--'));
      const VALID_MODES = new Set(['code-first', 'plan-first', 'hybrid']);
      if (!modeValue || !VALID_MODES.has(modeValue)) {
        error('Usage: set-mode <code-first|plan-first|hybrid> [--phase N]');
      }
      const keyPath = named.phase ? `phase_modes.${named.phase}` : 'default_phase_mode';
      const { setConfigValue } = require('./lib/config.cjs');
      const result = setConfigValue(cwd, keyPath, modeValue);
      core.output(result, raw, `Mode set: ${keyPath} = ${modeValue}`);
      break;
    }

This uses `parseNamedArgs()` per CLAUDE.md convention. It calls `setConfigValue()` directly (which internally calls `isValidConfigKey()` -- the config.cjs patch makes this work). Per D-07, D-08, D-09.
